1. 对象拷贝

   ```javascript
   function extend(old,new){
     for(let key in old){
       if(new[key]){
         continue
       }else{
         new.key = old[key]
       }
     }
   }
   ```

   

2.原型继承

```javascript
function Person(){
  this.have = 119
}

function Student(num,name,age){
  this.num = num
  this.name = name
  this.age = age
}

Student.prototype = new Person()
Student.prototype.constructor = Student

var wo = new Student(1,'周雨',18)
console.log(wo)
console.log(wo.__proto__)
console.log(wo.have)
```

3.组合继承:也造成了原型链

```javascript
function Person(name, age) {
      this.name = name
      this.age = age
      this.have = 119
      this.sayHi = function () {
        console.log("hello")
      }
    }

    function Student(num, name, age) {
      Person.call(this, name, age)	//继承属性	
      this.num = num
    }

    Student.prototype = new Person()			//继承方法
    Student.prototype.constructor = Student

    var wo = new Student(1, '周雨', 18)
    console.log(wo)
    console.log(wo.sayHi())
```
